1. Introduction to SQL Joins

Inner Join

SELECT *
FROM "especies" AS "es"
JOIN "animales" AS "an" ON "an".especie_id = "es".id;

Left Join

SELECT *
FROM "especies" AS "es"
LEFT JOIN "animales" AS "an" ON "an".especie_id = "es".id;

Right Join

SELECT *
FROM "especies" AS "es"
RIGHT JOIN "animales" AS "an" ON "an".especie_id = "es".id;

Full Join

SELECT *
FROM "especies" AS "es"
FULL JOIN "animales" AS "an" ON "an".especie_id = "es".id;

BONUS (CROSS JOIN)

SELECT *
FROM "especies" AS "es"
CROSS JOIN "animales" AS "an"
